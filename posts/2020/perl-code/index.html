Perl code example

use strictures 2;
use DBIx::Mint;
use Blog::Schema;
# use Devel::KYTProf;

use v5.10;

DBIx::Mint->connect( 'dbi:SQLite:dbname=blog.db', '', '',
                     { AutoCommit => 1, RaiseError => 1 } );

my $latest_entry
    = Blog::Entry->result_set->search->order_by({ -desc => 'id' })->single;
my $id = $latest_entry ? $latest_entry->{id} : 0;
$id++;

Blog::Entry->insert(
    id => $id,
    title => 'Hello world',
    body => 'It is rainy today.'
);
Blog::Comment->insert(
    {entry_id => $id, body => 'You are crazy.'},
    { entry_id => $id, body => 'You are funny.' },
    { entry_id => $id, body => 'You are weird.' },
);

my $entry = Blog::Entry->find($id);

say 'entry title: ' . $entry->title;
say 'entry body: ' . $entry->body;

my @comments = $entry->get_comments;
for my $comment (@comments) {
    say 'comment body: ' . $comment->body . ' for entry:' . $comment->get_entry->id;
}
